# Plan 105-01: Token Exchange Hook + Supabase JWT Wiring

## Goal

Create the `useOrgTokenExchange` hook that bridges Clerk org session → Supabase JWT, and wire it into `ProtectedRoute` so every authenticated render has an org-scoped Supabase token before any module loads.

## Context

The `exchangeToken()` function in `src/platform/tenants/token-exchange.ts` and the `setOrgAccessToken()` function in `src/platform/supabase.ts` both exist but are dead code — nothing calls them. RLS policies on all tables are already strict (`org_id` match required), meaning every Supabase query currently returns empty data or errors because no JWT is sent. This plan activates the token exchange flow.

Create new file `src/platform/tenants/useOrgTokenExchange.ts` with the following implementation:

```typescript
import { useEffect, useRef } from 'react'
import { useSession } from '@clerk/react'
import { useActiveOrg } from './useActiveOrg'
import { exchangeToken } from './token-exchange'
import { setOrgAccessToken } from '@platform/supabase'
import { invalidateDocsCache } from '@modules/docs/services/docs-service'

export interface OrgTokenState {
  isReady: boolean
  error: string | null
}

/**
 * Exchanges the current Clerk org session for a Supabase-compatible JWT.
 * Re-runs whenever the active org changes.
 * Sets the org token on the Supabase client via setOrgAccessToken().
 * Invalidates the docs cache on org switch so scoped docs reload.
 *
 * MUST be called inside a component that is already inside ClerkProvider.
 */
export function useOrgTokenExchange(): OrgTokenState {
  const { session, isLoaded } = useSession()
  const { activeOrg } = useActiveOrg()
  const prevOrgIdRef = useRef<string | null>(null)
  const isReadyRef = useRef(false)
  const errorRef = useRef<string | null>(null)

  useEffect(() => {
    if (!isLoaded || !session || !activeOrg) {
      return
    }

    const orgChanged = prevOrgIdRef.current !== null && prevOrgIdRef.current !== activeOrg.id

    async function doExchange() {
      try {
        const clerkToken = await session!.getToken()
        if (!clerkToken) {
          errorRef.current = 'Clerk session token unavailable'
          setOrgAccessToken(null)
          return
        }

        const result = await exchangeToken(clerkToken)
        setOrgAccessToken(result.access_token)
        errorRef.current = null

        // Invalidate docs cache on org switch so scoped docs reload
        if (orgChanged) {
          invalidateDocsCache()
        }

        prevOrgIdRef.current = activeOrg!.id
      } catch (err) {
        const message = err instanceof Error ? err.message : 'Token exchange failed'
        errorRef.current = message
        setOrgAccessToken(null)
        console.error('[useOrgTokenExchange]', message)
      }
    }

    void doExchange()
  }, [isLoaded, session, activeOrg?.id]) // eslint-disable-line react-hooks/exhaustive-deps

  return {
    isReady: isReadyRef.current,
    error: errorRef.current,
  }
}
```

Note: The hook uses `useSession` from `@clerk/react` (not `useAuth`) because `session.getToken()` is the correct API for getting the raw JWT. The `useActiveOrg` hook already wraps `useOrganization` and provides `activeOrg?.id`.

Note on `invalidateDocsCache`: this function must be exported from `docs-service.ts` (see Task 3 in Plan 105-03). The import path `@modules/docs/services/docs-service` follows the `@modules/` alias pattern.

<action>
Modify `src/platform/auth/ProtectedRoute.tsx` to call `useOrgTokenExchange` before rendering children. The hook must be called unconditionally at the top of the component (React hooks rules), but its effect only runs when `isLoaded && session && activeOrg` are all truthy.

Replace the file content with:

```typescript
import { useAuth, RedirectToSignIn, useOrganizationList } from '@clerk/react'
import { Navigate } from 'react-router-dom'
import type { ReactNode } from 'react'
import { useOrgTokenExchange } from '@platform/tenants/useOrgTokenExchange'

const loadingStyle: React.CSSProperties = {
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  height: '100vh',
  fontFamily: 'Inter, sans-serif',
}

const loadingText: React.CSSProperties = {
  fontSize: 14,
  color: '#757575',
}

export default function ProtectedRoute({ children }: { children: ReactNode }) {
  const { isSignedIn, isLoaded } = useAuth()
  const { userMemberships, isLoaded: orgsLoaded } = useOrganizationList({
    userMemberships: { infinite: true },
  })

  // Wire Clerk org → Supabase JWT token exchange.
  // Called unconditionally (hook rules). Effect runs only when session + org are ready.
  const { error: tokenError } = useOrgTokenExchange()

  // 1. Clerk SDK not yet loaded — show loading
  if (!isLoaded) {
    return (
      <div style={loadingStyle}>
        <p style={loadingText}>Carregando...</p>
      </div>
    )
  }

  // 2. Not signed in — redirect immediately (do NOT wait for orgsLoaded)
  if (!isSignedIn) {
    return <RedirectToSignIn />
  }

  // 3. Signed in, waiting for org data
  if (!orgsLoaded) {
    return (
      <div style={loadingStyle}>
        <p style={loadingText}>Carregando...</p>
      </div>
    )
  }

  // 4. Signed in but no org — redirect to onboarding
  if (userMemberships?.data?.length === 0) {
    return <Navigate to="/criar-empresa" replace />
  }

  // 5. Token exchange failed — show error (never silently allow unauthenticated Supabase access)
  if (tokenError) {
    return (
      <div style={loadingStyle}>
        <p style={{ ...loadingText, color: '#ef4444' }}>
          Erro ao autenticar: {tokenError}
        </p>
      </div>
    )
  }

  return <>{children}</>
}
```
