function relativeDate(ts: string): string {
  const d = new Date(ts)
  const now = new Date()
  const diffMs = now.getTime() - d.getTime()
  const diffDays = Math.floor(diffMs / (1000 * 60 * 60 * 24))
  if (diffDays === 0) return 'Hoje'
  if (diffDays === 1) return 'Ontem'

**1. Update `src/modules/tasks/manifest.ts`**

Add the `extensions` array to the tasks manifest. Import `RecentTasksWidget` from the local extensions directory and reference SLOT_IDS from the registry:

```typescript
import { RecentTasksWidget } from './extensions/RecentTasksWidget'
import { SLOT_IDS } from '@/modules/registry'
// MODULE_IDS already imported for the manifest id field (from Phase 38)
```

Add to the manifest object:
```typescript
extensions: [
  {
    id: 'tasks-home-widget',
    description: 'Mostra tarefas pendentes recentes no Home dashboard',
    requires: [MODULE_IDS.TASKS],
    injects: {
      [SLOT_IDS.HOME_DASHBOARD]: RecentTasksWidget,
    },
  },
],
```

Note: `requires: [MODULE_IDS.TASKS]` means this extension only renders when the Tasks module itself is enabled. This is the simplest valid requires[] — the extension depends on its own module.
