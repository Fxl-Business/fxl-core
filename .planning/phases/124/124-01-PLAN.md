2. **Define mintJWT(userId, orgId, secret)**: Mint a Supabase-compatible JWT using pure Node.js
   (`node:crypto` — no new dependencies needed). Algorithm: HS256.
   Payload: `{ sub: userId, org_id: orgId, role: 'authenticated', aud: 'authenticated', iss: 'supabase', iat: now, exp: now + 3600 }`.
   Header: `{ alg: 'HS256', typ: 'JWT' }`.
   Implementation: base64url-encode header + payload, HMAC-SHA256 the concatenation with the secret,
   base64url-encode the signature, join with dots.

   ```typescript
   import { createHmac } from 'node:crypto'

   function base64url(input: string): string {
     return Buffer.from(input).toString('base64url')
   }

   function mintJWT(userId: string, orgId: string, secret: string): string {
     const now = Math.floor(Date.now() / 1000)
     const header = base64url(JSON.stringify({ alg: 'HS256', typ: 'JWT' }))
     const payload = base64url(JSON.stringify({
       sub: userId,
       org_id: orgId,
       role: 'authenticated',
       aud: 'authenticated',
       iss: 'supabase',
       iat: now,
       exp: now + 3600,
     }))
     const signingInput = `${header}.${payload}`
     const signature = createHmac('sha256', secret)
       .update(signingInput)
       .digest('base64url')
     return `${signingInput}.${signature}`
   }
   ```

3. **Define queryDocuments(jwt, supabaseUrl, anonKey)**: Query documents table using a
   Supabase client that injects the org JWT. Use `@supabase/supabase-js` (already in deps).
   Create a fresh client per call (no shared singleton state between org A and org B checks):

   ```typescript
   import { createClient } from '@supabase/supabase-js'

   async function queryDocuments(jwt: string, url: string, anonKey: string): Promise<number> {
     const client = createClient(url, anonKey, {
       global: {
         fetch: (input, init) => {
           const headers = new Headers(init?.headers)
           headers.set('Authorization', `Bearer ${jwt}`)
           headers.set('apikey', anonKey)
           return fetch(input, { ...init, headers })
         },
       },
     })
     const { data, error } = await client.from('documents').select('id', { count: 'exact', head: false })
     if (error) throw new Error(`Supabase query error: ${error.message}`)
     return data?.length ?? 0
   }
   ```

4. **Define checkEdgeFunctionHealth(functionsUrl)**: Send a POST with a known-bad
   Authorization header to `auth-token-exchange`. Expect a 401 response (NOT a CORS error or
   network failure). A CORS error means the function is not deployed (Pitfall #10).

   ```typescript
   async function checkEdgeFunctionHealth(functionsUrl: string): Promise<void> {
     const res = await fetch(`${functionsUrl}/auth-token-exchange`, {
       method: 'POST',
       headers: {
         Authorization: 'Bearer invalid-token-for-health-check',
         'Content-Type': 'application/json',
       },
       body: JSON.stringify({ org_id: 'test' }),
     })
     // We expect 401 (invalid token) — that proves the function is deployed and running
     // 404 or network error = not deployed
     if (res.status === 401) return // pass
     const body = await res.text().catch(() => '')
     throw new Error(`auth-token-exchange health check failed: expected 401, got ${res.status}. Body: ${body.slice(0, 200)}`)
   }
   ```
