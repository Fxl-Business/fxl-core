**Google OAuth handler:**
```typescript
async function handleGoogleOAuth() {
  if (!signIn) return
  setOauthLoading(true)
  setError('')
  try {
    await signIn.authenticateWithRedirect({
      strategy: 'oauth_google',
      redirectUrl: '/sso-callback',
      redirectUrlComplete: '/',
    })
  } catch (err) {
    setOauthLoading(false)
    if (isClerkAPIResponseError(err)) {
      setError(err.errors[0]?.message || 'Erro ao conectar com Google')
    } else {
      setError('Erro ao conectar com Google. Tente novamente.')
    }
  }
}
```

**Email/password handler:**
```typescript
async function handleEmailPassword(e: React.FormEvent) {
  e.preventDefault()
  if (!signIn) return
  setLoading(true)
  setError('')
  try {
    const result = await signIn.create({ identifier: email, password })
    if (result.status === 'complete') {
      await setActive({ session: result.createdSessionId })
      navigate('/')
    } else if (result.status === 'needs_second_factor') {
      // MFA not custom-handled in this phase — Clerk handles it
      setError('Autenticacao de dois fatores necessaria. Use o login padrao.')
    } else {
      setError('Verificacao adicional necessaria.')
    }
  } catch (err) {
    if (isClerkAPIResponseError(err)) {
      setError(err.errors[0]?.message || 'Email ou senha incorretos')
    } else {
      setError('Erro ao fazer login. Tente novamente.')
    }
  } finally {
    setLoading(false)
  }
}
```
