    **1. Create `src/platform/hooks/useAdminMode.ts`:**

    Hook that exposes admin state for use across the app:

    ```typescript
    import { useAuth } from '@clerk/react'
    import { useLocation, useNavigate } from 'react-router-dom'
    import { useCallback } from 'react'

    export interface AdminModeState {
      isSuperAdmin: boolean
      isAdminRoute: boolean
      toggleMode: () => void
    }

    export function useAdminMode(): AdminModeState {
      const { sessionClaims } = useAuth()
      const location = useLocation()
      const navigate = useNavigate()

      const isSuperAdmin = Boolean(
        (sessionClaims as Record<string, unknown> | undefined)?.super_admin
      )
      const isAdminRoute = location.pathname.startsWith('/admin')

      const toggleMode = useCallback(() => {
        if (isAdminRoute) {
          navigate('/')
        } else {
          navigate('/admin')
        }
      }, [isAdminRoute, navigate])

      return { isSuperAdmin, isAdminRoute, toggleMode }
    }
    ```

    The `sessionClaims` access pattern uses Clerk's JWT custom claims. The `super_admin` claim is set via Clerk JWT template (Phase 75). Cast to `Record<string, unknown>` because Clerk's type for sessionClaims is `CustomJwtSessionClaims` which may not include custom fields without type augmentation.

    **2. Create `src/platform/auth/SuperAdminRoute.tsx`:**

    Stub that Phase 75 will finalize. If Phase 75 already created this file, skip this step:

    ```typescript
    import { useAuth, RedirectToSignIn } from '@clerk/react'
    import { Navigate } from 'react-router-dom'
    import type { ReactNode } from 'react'

    export default function SuperAdminRoute({ children }: { children: ReactNode }) {
      const { isSignedIn, isLoaded, sessionClaims } = useAuth()

      if (!isLoaded) {
        return (
          <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: '100vh', fontFamily: 'Inter, sans-serif' }}>
            <p style={{ fontSize: 14, color: '#757575' }}>Carregando...</p>
          </div>
        )
      }

      if (!isSignedIn) {
        return <RedirectToSignIn />
      }

      const isSuperAdmin = Boolean(
        (sessionClaims as Record<string, unknown> | undefined)?.super_admin
      )

      if (!isSuperAdmin) {
        return <Navigate to="/" replace />
      }

      return <>{children}</>
    }
    ```
