**ExtensionProvider:**
```typescript
export function ExtensionProvider({ children }: { children: React.ReactNode }) {
  const enabledIds = useEnabledModuleIds()
  const enabledSet = useMemo(() => new Set(enabledIds), [enabledIds])
  const map = useMemo(
    () => resolveExtensions(MODULE_REGISTRY, enabledSet),
    [enabledSet],
  )
  return <ExtensionContext.Provider value={map}>{children}</ExtensionContext.Provider>
}
```

**ExtensionSlot component** (CONT-03):
```typescript
export function ExtensionSlot({
  id,
  context,
  className
}: {
  id: string
  context?: Record<string, string | number | boolean>
  className?: string
}) {
  const map = useContext(ExtensionContext)
  const components = map.get(id)
  if (!components || components.length === 0) return null
  return (
    <>
      {components.map((Component, i) => (
        <Component key={i} context={context} className={className} />
      ))}
    </>
  )
}
```

Key constraints:
- ExtensionSlot MUST return null when no extensions are registered for the slot ID (not an empty div, not a fragment with nothing). This is explicit in success criteria 3 of the phase.
- ExtensionSlot passes `context` and `className` from its own props down to each registered component — these match `SlotComponentProps`.
- The context used is React's createContext — not Zustand, not any state library.
- useMemo dependency on enabledIds ensures resolveExtensions re-runs only when modules are toggled.
- Do NOT use `React.lazy()` for slot-registered components — they are always bundled (Performance Trap from research).

**Part A — Create `src/modules/hooks/useActiveExtensions.ts`:**

This convenience hook returns the active extensions for a specific module. It is a thin wrapper over the ExtensionContext that filters by module ID.

```typescript
import { MODULE_REGISTRY } from '../registry'
import type { ModuleExtension } from '../registry'
import { useExtensions } from '../slots'

/**
 * Returns the active extensions for a given module ID.
 * An extension is "active" if:
 * 1. The module providing the extension is enabled
 * 2. All modules in the extension's requires[] are enabled
 *
 * @param moduleId - The module ID to get extensions for
 * @returns Array of active ModuleExtension objects for that module
 */
export function useActiveExtensions(moduleId: string): ModuleExtension[] {
  const extensionMap = useExtensions()

  // Find the module in the registry
  const mod = MODULE_REGISTRY.find(m => m.id === moduleId)
  if (!mod?.extensions) return []

  // An extension is active if at least one of its injected slots
  // appears in the resolved extension map with its component
  return mod.extensions.filter(ext => {
    return Object.keys(ext.injects).some(slotId => {
      const components = extensionMap.get(slotId)
      return components && components.length > 0
    })
  })
}
```

The hook returns `ModuleExtension[]` — the full extension objects (id, requires, description, injects) so consumers can display extension metadata (e.g., in the admin panel).

If the `src/modules/hooks/` directory does not exist, create it.
