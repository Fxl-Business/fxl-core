Replace the entire file content:

```typescript
import { getAllDocuments, type DocumentRow } from './docs-service'

export type SearchEntry = {
  title: string
  description?: string
  badge?: string
  href: string
}

export async function buildSearchIndex(): Promise<SearchEntry[]> {
  const docs = await getAllDocuments()
  return docs.map((doc: DocumentRow) => ({
    title: doc.title,
    description: doc.description ?? undefined,
    badge: doc.badge ?? undefined,
    href: `/${doc.slug}`,
  }))
}
```

Key changes:
- Import from `docs-service` instead of `docs-parser`
- `buildSearchIndex()` is now `async` and returns `Promise<SearchEntry[]>`
- Maps `DocumentRow` fields directly to `SearchEntry` (no parsing needed — search only uses metadata)
- `href` is `/${slug}` — prepend slash to slug from database
- `description` and `badge` use nullish coalescing to convert `null` to `undefined` (matching SearchEntry type)
- The `SearchEntry` type stays exactly the same

This needs to become an async fetch. The best approach: load the search index when the dialog opens (lazy loading), since the search dialog is opened on-demand (Cmd+K).

Updated approach:

```typescript
const [index, setIndex] = useState<SearchEntry[]>([])
const [indexLoaded, setIndexLoaded] = useState(false)

function handleOpenChange(next: boolean) {
  setOpen(next)
  if (next === true) {
    // Load search index on first open
    if (!indexLoaded) {
      buildSearchIndex().then((entries) => {
        setIndex(entries)
        setIndexLoaded(true)
      }).catch(() => {})
    }
    // Load KB entries (existing behavior)
    if (kbEntries.length === 0) {
      listKnowledgeEntries({}).then(setKbEntries).catch(() => {})
    }
  }
  if (next === false) {
    setQuery('')
  }
}
```
